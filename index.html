/** * Bible GongBoo Quiz API * * This is a new, isolated Apps Script project for the Bible spreadsheet. * It does not modify or depend on the existing "Test of gongboo" project. * Authentication tokens are issued by the existing "Test of member" project. */ const BIBLE_SPREADSHEET_ID = '1-qdMFaDKjVsz8vQmtkeTydWZPDOl0HT8HTwjcMZULas'; const BIBLE_TRIAL_LIMIT = 20; const BIBLE_DEFAULT_LIMIT = 120; const BIBLE_MAX_LIMIT = 200; const BIBLE_SHEETS = { 'bible-ot': { code: 'BIBLE_OT', name: 'Old Testament · 구약', category: 'BIBLE', sheet: 'BIBLE-OT' }, 'bible-nt': { code: 'BIBLE_NT', name: 'New Testament · 신약', category: 'BIBLE', sheet: 'BIBLE-NT' } }; function doGet(e) { try { return handleBibleRequest_((e && e.parameter) || {}); } catch (error) { console.error(error); return bibleJson_({ status: 'error', code: error && error.publicCode ? error.publicCode : 'REQUEST_FAILED', message: error && error.publicMessage ? error.publicMessage : 'The Bible question request could not be completed.' }); } } function doPost(e) { try { const body = e && e.postData && e.postData.contents ? JSON.parse(e.postData.contents) : {}; return handleBibleRequest_(body || {}); } catch (error) { console.error(error); return bibleJson_({ status: 'error', code: error && error.publicCode ? error.publicCode : 'REQUEST_FAILED', message: error && error.publicMessage ? error.publicMessage : 'The Bible question request could not be completed.' }); } } function handleBibleRequest_(params) { const action = bibleText_(params.action).toLowerCase(); if (action === 'health') { return bibleJson_({ status: 'success', service: 'Bible GongBoo Quiz API' }); } if (action === 'subjects') { return bibleJson_({ status: 'success', subjects: Object.keys(BIBLE_SHEETS).map(function (key) { return bibleSubjectInfo_(BIBLE_SHEETS[key]); }) }); } const access = bibleVerifyAccessToken_(params.session_token); const config = bibleResolveSheet_(params.sheet); let start = Math.max(1, parseInt(params.start, 10) || 1); let limit = Math.max( 1, Math.min(parseInt(params.limit, 10) || BIBLE_DEFAULT_LIMIT, BIBLE_MAX_LIMIT) ); if (access.status === 'p' && access.role !== 'admin') { start = 1; limit = BIBLE_TRIAL_LIMIT; } const sheet = bibleGetSheet_(config.sheet); const availableTotal = Math.max(0, sheet.getLastRow() - 1); const total = access.status === 'p' && access.role !== 'admin' ? Math.min(BIBLE_TRIAL_LIMIT, availableTotal) : availableTotal; if (String(params.total || '').toLowerCase() === 'true') { return bibleJson_({ status: 'success', total: total, sheet: config.sheet }); } return bibleJson_({ status: 'success', data: bibleReadQuestions_(sheet, start, limit, availableTotal), total: total, start: start, limit: limit, sheet: config.sheet }); } function bibleSubjectInfo_(config) { return { CODE: config.code, NAME: config.name, CATEGORY: config.category, SHEET: config.sheet, SET_SIZE: 120, FORMAT: 'BIBLE_EN_KO', ACTIVE: 'Y', VERSION: '1.0' }; } function bibleResolveSheet_(requested) { const key = bibleText_(requested || 'BIBLE-OT').toLowerCase(); const config = BIBLE_SHEETS[key]; if (!config) { biblePublicError_('SHEET_NOT_ALLOWED', 'The requested Bible section is unavailable.'); } return config; } function bibleGetSheet_(sheetName) { const spreadsheet = SpreadsheetApp.openById(BIBLE_SPREADSHEET_ID); const sheet = spreadsheet.getSheetByName(sheetName); if (!sheet) { biblePublicError_('SHEET_NOT_FOUND', 'The Bible question sheet was not found.'); } return sheet; } function bibleReadQuestions_(sheet, start, limit, availableTotal) { if (availableTotal < 1 || start > availableTotal) return []; const lastColumn = Math.max(1, sheet.getLastColumn()); const headers = sheet.getRange(1, 1, 1, lastColumn) .getDisplayValues()[0].map(function (header) { return bibleText_(header); }); const rowCount = Math.min(limit, availableTotal - start + 1); const values = sheet.getRange(start + 1, 1, rowCount, lastColumn) .getDisplayValues(); return values.filter(function (row) { return bibleText_(row[0]) !== ''; }).map(function (row) { const record = {}; headers.forEach(function (header, index) { if (header) record[header] = row[index] == null ? '' : row[index]; }); return record; }); } function bibleVerifyAccessToken_(token) { const value = bibleText_(token); if (!value) biblePublicError_('AUTH_REQUIRED', 'Please log in again.'); const parts = value.split('.'); if (parts.length !== 2) biblePublicError_('AUTH_INVALID', 'Please log in again.'); const secret = PropertiesService.getScriptProperties() .getProperty('GONGBOO_AUTH_SHARED_SECRET'); if (!secret || secret.length < 32) { throw new Error('GONGBOO_AUTH_SHARED_SECRET is missing or too short.'); } const expected = bibleBase64UrlBytes_( Utilities.computeHmacSha256Signature(parts[0], secret) ); if (!bibleConstantTimeEqual_(expected, parts[1])) { biblePublicError_('AUTH_INVALID', 'Please log in again.'); } let payload; try { // Test of member signs an encodeURIComponent(JSON) payload, not Base64. payload = JSON.parse(decodeURIComponent(parts[0])); } catch (error) { biblePublicError_('AUTH_INVALID', 'Please log in again.'); } const now = Math.floor(Date.now() / 1000); if (!payload || !payload.email || !payload.exp || payload.exp <= now) { biblePublicError_('AUTH_INVALID', 'Please log in again.'); } payload.status = bibleText_(payload.status).toLowerCase(); payload.role = bibleText_(payload.role).toLowerCase(); if ( payload.role !== 'admin' && payload.status !== 'a' && payload.status !== 'p' ) { biblePublicError_('ACCESS_DENIED', 'This account cannot access Bible questions.'); } return payload; } function bibleAddBase64Padding_(value) { let result = String(value || ''); while (result.length % 4) result += '='; return result; } function bibleBase64UrlBytes_(bytes) { return Utilities.base64EncodeWebSafe(bytes).replace(/=+$/g, ''); } function bibleConstantTimeEqual_(left, right) { left = String(left || ''); right = String(right || ''); let mismatch = left.length ^ right.length; const length = Math.max(left.length, right.length); for (let i = 0; i < length; i += 1) { mismatch |= (left.charCodeAt(i) || 0) ^ (right.charCodeAt(i) || 0); } return mismatch === 0; } function bibleText_(value) { return String(value == null ? '' : value).trim(); } function biblePublicError_(code, message) { const error = new Error(message); error.publicCode = code; error.publicMessage = message; throw error; } function bibleJson_(payload) { return ContentService .createTextOutput(JSON.stringify(payload)) .setMimeType(ContentService.MimeType.JSON); }
